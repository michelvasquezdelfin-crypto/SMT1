1 Abre tu Google Sheet
Ve al Sheet PkgScan — Historial de Paquetes en tu Google Drive.
2 Abre Apps Script
En el menú: Extensiones → Apps Script
3 Pega este código
function doPost(e) { const sheet = SpreadsheetApp .getActiveSpreadsheet() .getActiveSheet(); // Encabezados si la hoja está vacía if (sheet.getLastRow() === 0) { sheet.appendRow([ 'Fecha Escaneo', 'Origen (Cuadro Negro)', 'ID / Guía Paquete', 'Fecha Promesa', 'Confianza IA (%)', 'Notas' ]); // Formato encabezados sheet.getRange(1,1,1,6) .setBackground('#1a1a1a') .setFontColor('#00e5ff') .setFontWeight('bold'); } const data = JSON.parse(e.postData.contents); sheet.appendRow([ data.fecha || '', data.origen || 'N/D', data.id_paquete|| 'N/D', data.fecha_promesa || 'N/D', data.confianza || 0, data.notas || '' ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'PkgScan OK'})) .setMimeType(ContentService.MimeType.JSON); }
